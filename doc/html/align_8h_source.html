140 template<typename M>
141 std::string
142 Alignment::build_string(typename Polymer<M>::const_iterator q_lo,
143  typename Polymer<M>::const_iterator q_hi) const {
144  const size_t q_size = q_hi - q_lo;
145  const size_t t_size = traceback.cols()-1;
146 
147  assert (q_size+1 == traceback.rows());
148 
149  std::string aligned_query;
150 
151  size_t i = q_size, j = t_size;
152  while (i + j != 0) {
153  switch (traceback.elem(i, j).move) {
154  case Cell::Move::GAP_Q:
155  aligned_query.push_back(gap_char<M>());
156  --j;
157  break;
158  case Cell::Move::GAP_T:
159  aligned_query.push_back(ins_char<M>(q_lo[i-1]));
160  --i;
161  break;
162  default:
163  aligned_query.push_back(reg_char<M>(q_lo[i-1]));
164  --i;
165  --j;
166  }
167  }
168  std::reverse(aligned_query.begin(), aligned_query.end());
169 
170  return aligned_query;
171 }

209 template<typename M>
210 void
211 nw_align(typename Polymer<M>::const_iterator q_lo,
212  typename Polymer<M>::const_iterator q_hi,
213  typename Polymer<M>::const_iterator t_lo,
214  typename Polymer<M>::const_iterator t_hi,
215  const Matrix<int32_t> &match,
216  int32_t gapp,
217  Alignment &result,
218  bool score_only=false) {
219  result.clear();
220 
221  const size_t q_size = q_hi - q_lo;
222  const size_t t_size = t_hi - t_lo;
223  thread_local Matrix<Cell> &trace = result.traceback;
224  trace.resize(q_size+1, t_size+1);
225 
226  for (int i = 1; i < trace.rows(); ++i) {
227  trace.elem(i, 0).score = -gapp * i;
228  trace.elem(i, 0).move = Cell::Move::GAP_T; //Cell::Move::GAP_B;
229  }
230  for (int j = 1; j < trace.cols(); ++j) {
231  trace.elem(0, j).score = -gapp * j;
232  trace.elem(0, j).move = Cell::Move::GAP_Q; //Cell::Move::GAP_A;
233  }
234 
235  for (size_t i = 0; i < q_size; ++i) {
236  size_t n = q_lo[i].index();
237  for (size_t j = 0; j < t_size; ++j) {
238  size_t m = t_lo[j].index();
239 
240  Cell cell;
241  cell.move = Cell::Move::MATCH;
242  cell.score = trace.elem(i, j).score + match.elem(m, n);
243 
244  int32_t gappa_score = trace.elem(i+1, j).score - gapp;
245  if (gappa_score > cell.score) {
246  cell.score = gappa_score;
247  cell.move = Cell::Move::GAP_Q;
248  }
249 
250  int32_t gappb_score = trace.elem(i, j+1).score - gapp;
251  if (gappb_score > cell.score) {
252  cell.score = gappb_score;
253  cell.move = Cell::Move::GAP_T;
254  }
255 
256  trace.elem(i+1, j+1) = cell;
257  }
258  }
259  result.score = trace.elem(q_size, t_size).score;
260 
261  if (!score_only) result.aligned_query = result.build_string<M>(q_lo, q_hi);
262 }
